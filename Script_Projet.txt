    DROP TABLE Projet.Products;

-- Création de la table Products
CREATE TABLE [Projet].[Products]
(      
    ProductID INT IDENTITY(1,1) PRIMARY KEY, -- Utilisation de INT IDENTITY pour auto-incrémenter
    ProductName VARCHAR(50),
    SupplierID INT,
    CategoryID INT,
    Unit VARCHAR(25),
    Price DECIMAL(18, 2), -- Utilisation de DECIMAL pour le prix avec précision
    FOREIGN KEY (CategoryID) REFERENCES Projet.Categories (CategoryID),
    FOREIGN KEY (SupplierID) REFERENCES Projet.Suppliers (SupplierID)
);

-- Création de la table OrderDate
CREATE TABLE [Projet].[OrderDate]
(
    OrderDateID INT IDENTITY(1,1) PRIMARY KEY, -- Colonne auto-incrémentée pour l'ID
    DateTime DATETIME,                         -- Stockage de la date et heure complète
    Date DATE,                                 -- Stockage de la date uniquement
    Year INT,                                  -- Année
    Quarter INT,                               -- Trimestre
    Month VARCHAR(20),                         -- Nom complet du mois (ex: September)
    Month_Ab CHAR(4),                          -- Abréviation du mois (ex: Sept)
    Month_N INT,                               -- Numéro du mois (ex: 9 pour September)
    Day INT                                    -- Jour du mois
);
